% ==================== Slide 3: Introduction ====================
\section{Introduction}
\begin{frame}{Introduction \& Background}
\begin{itemize}
    \item \textit{Salmonella} Enteritidis (SE) is a leading cause of foodborne illness worldwide
    \item Shell eggs are a major vehicle --- SE penetrates shell and grows rapidly in iron-rich yolk
    \item USDA-FSIS requires eggs stored $\leq 7.2\,^\circ$C within 12\,h of laying
    \item Temperature fluctuates during cooling, storage, and distribution
    \item \textbf{Objective:} Develop a dynamic Gompertz model for SE growth under sinusoidal temperature (3--43\,$^\circ$C, 24\,h cycle)
    \item Reference: Gumudavelli et al.\ (2007), \textit{J.\ Food Sci.}\ 72:M254--62
\end{itemize}
\end{frame}

% ==================== Slide 4: Model ====================
\section{Model}
\begin{frame}{Model Description}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Primary Model (Gompertz)}

Constant $T$ (analytical):
\begin{equation*}
y(t) = A + C \cdot e^{-e^{(\mu e / C)(M-t)+1}}
\end{equation*}

Variable $T$ (ODE):
\begin{equation*}
\frac{dy}{dt} = -\mu(T)\, e\, \frac{y-A}{C}\, \ln\!\left(\frac{y-A}{C}\right)
\end{equation*}

$y = \log_{10}N$;\; $A$: initial;\; $C$: range;\; $M$: inflection time
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Secondary Model (Ratkowsky)}
\begin{equation*}
\mu(T) = a(T - T_{\min})^2 \left(1 - e^{b(T - T_{\max})}\right)
\end{equation*}

$a, b$: regression coefficients\\
$T_{\min} = 6\,^\circ$C,\; $T_{\max} = 46.3\,^\circ$C (fixed)

\vspace{1em}
\textbf{Dynamic Model}\\
Integrate primary + secondary,\\
solve numerically with \texttt{ode45}.\\
Temperature interpolated via \texttt{interp1}.
\end{column}
\end{columns}
\end{frame}

% ==================== Slide 13: Summary ====================
\section{Summary}
\begin{frame}{Summary \& Conclusions}
\begin{itemize}
    \item Dynamic Gompertz model successfully developed for SE growth in egg yolk
    \item 5 parameters estimated, 2 fixed; RMSE $< 0.3\;\log_{10}$ CFU/mL
    \item All 5 standard statistical assumptions evaluated
    \item Optimal experimental design: key window at 2--8\,hr
    \item Bootstrap CIs consistent with asymptotic CIs
    \item \textbf{Practical significance:} Model predicts SE risk during egg cooling, storage, and distribution --- supports USDA-FSIS 7.2\,$^\circ$C recommendation
\end{itemize}
\end{frame}
